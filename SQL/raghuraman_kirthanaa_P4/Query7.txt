SELECT FOOD_DES.NDB_No, NUT_DATA.Nutr_No, NUT_DATA.Nutr_Val FROM FOOD_DES JOIN NUT_DATA ON(FOOD_DES.NDB_No = NUT_DATA.NDB_No)
WHERE
(
    (FOOD_DES.NDB_No IN (SELECT NDB_No FROM NUT_DATA WHERE Nutr_No = (SELECT Nutr_No FROM NUTR_DEF WHERE NutrDesc = 'Caffeine'))
    AND
    FOOD_DES.NDB_No IN (SELECT NDB_No FROM NUT_DATA WHERE Nutr_No = (SELECT Nutr_No FROM NUTR_DEF WHERE NutrDesc = 'Alcohol, ethyl')))
    AND
    (NUT_DATA.Nutr_No = (SELECT Nutr_No FROM NUTR_DEF WHERE NutrDesc = 'Caffeine')
    OR
    NUT_DATA.Nutr_No = (SELECT Nutr_No FROM NUTR_DEF WHERE NutrDesc = 'Alcohol, ethyl'))
);
